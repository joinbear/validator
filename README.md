##常用正则整理和收集
```
	//手机号
	phone: /^1[345789]\d{9}$/,
	//包含中文
	cn: /[\u4E00-\u9FA5]+/,
	//车牌号
	car: /^[京津沪渝冀豫云辽黑湘皖鲁新苏浙赣鄂桂甘晋蒙陕吉闽贵粤青藏川宁琼使领A-Z]{1}[A-Z]{1}[A-Z0-9]{4}[A-Z0-9挂学警港澳]{1}$/,
	//微信号，6至20位，以字母开头，字母，数字，减号，下划线
	wx: /^[a-zA-Z]([-_a-zA-Z0-9]{5,19})$/,
	//QQ号，5至11位
	qq:  /^[1-9][0-9]{4,10}$/,
	//日期
	day: /^(?:(?!0000)[0-9]{4}-(?:(?:0[1-9]|1[0-2])-(?:0[1-9]|1[0-9]|2[0-8])|(?:0[13-9]|1[0-2])-(?:29|30)|(?:0[13578]|1[02])-31)|(?:[0-9]{2}(?:0[48]|[2468][048]|[13579][26])|(?:0[48]|[2468][048]|[13579][26])00)-02-29)$/,
	//RGB Hex颜色
	color: /^#?([a-fA-F0-9]{6}|[a-fA-F0-9]{3})$/,
	// ipv4
	ipv4: /^(?:(?:25[0-5]|2[0-4][0-9]|[01]?[0-9][0-9]?)\.){3}(?:25[0-5]|2[0-4][0-9]|[01]?[0-9][0-9]?)$/,
	//身份证号
	identity: /^[1-9]\d{7}((0\d)|(1[0-2]))(([0|1|2]\d)|3[0-1])\d{3}$|^[1-9]\d{5}[1-9]\d{3}((0\d)|(1[0-2]))(([0|1|2]\d)|3[0-1])\d{3}([0-9]|X|x)$/,
	// Email
	email: /^([A-Za-z0-9_\-\.])+\@([A-Za-z0-9_\-\.])+\.([A-Za-z]{2,4})$/,
	//正数
	positive: /^\d*\.?\d+$/,
	//负数
	negative: /^-\d*\.?\d+$/,
	//密码强度正则，最少6位，包括至少1个大写字母，1个小写字母，1个数字，1个特殊字符
	password: /^.*(?=.{6,})(?=.*\d)(?=.*[A-Z])(?=.*[a-z])(?=.*[!@#$%^&*? ]).*$/,
	//用户名正则，4到16位（字母，数字，下划线，减号）
	username: /^[a-zA-Z0-9_-]{4,16}$/,
	//ios系统
	ios: /iPad|iPhone|iPod/ig
```
##使用
```
	$npm i https://github.com/joinbear/validator.git --save-dev
	const { validator , pattern , helper } = require('validator');
	const phone = '18212345678';
	// validate
	console.log(validator.isPhone(phone));// true
	// get pattern
	console.log(pattern.phone);// /^1[345789]\d{9}$/
	helper see ./src/helper.js
```
